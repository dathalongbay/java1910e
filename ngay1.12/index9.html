toán tử = ( phép gán )
        ví dụ : var i = 5;

        toán tử == ( so sánh giá trị )
        ví dụ : 3 == "3"
        toán tử === ( so sánh giá trị và kiểu dữ liệu )
        ví dụ : 3 === "3"

        toán tử so sánh khác !=
        ví dụ : 3 != "3"
        toán tử so sánh khác tuyệt đối !==
        ví dụ : 3 !== "3"

        toán tử tăng thêm 1 đơn vị
        var i = 5;
        i++; ( i sẽ tăng thêm 1 đơn vị và bằng 6 )

        toàn tử giảm 1 đơn vị
        var i = 5;
        i--;
