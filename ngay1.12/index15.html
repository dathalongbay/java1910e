1 số toàn tử rút gọn trong js
        ví dụ 1 :
        var a = 5;
        => cách 1 : a = a + 5;
        => cách 2 : a += 5;
        ví dụ 2 :
        var b = 7;
        => cách 1 : b = b - 3;
        => cách 3 : b -= 3;
        ví dụ 3 :
        var c = 3;
        => cách 1 : c = c * 3;
        => cách 2 : c *= 3;
        ví dụ 4 :
        var d = 7;
        => cách 1 : d = d/3;
        => cách 2 : d /= 3;
        ví dụ 5 :
        var e = 5;
        => cách 1 : e = e % 2;
        => cách 2 : e %= 2;
        ví dụ 6 : phép nối chuỗi rút gọn
        var f = "chào";
        cách 1 : f = f + " việt nam";
        cách 2 : f += "việt nam";
