Vòng lặp while
        Cú pháp :

        //sự khởi đầu của biến đếm
        while(điều kiện chạy của vòng lặp) {
            // các câu lệnh của vòng lặp

            // sự thay đổi của biến đếm
        }

        Ví dụ :
        var n = 1;
        while(n <= 10) {
            console.log("số thứ : " + n);

            n++;
        }
